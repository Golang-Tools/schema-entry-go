func ( reflect.Type) bool {
	if likePtrKind(.Kind()) {
		return true
	}
	if .Kind() == reflect.Struct {
		if .NumField() != 1 {
			return false
		}
		return (.Field(0).Type)
	}
	if .Kind() == reflect.Array {
		if .Len() != 1 {
			return false
		}
		return (.Elem())
	}
	return false
}
