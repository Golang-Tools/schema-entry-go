func ( *subSchema) ( *subSchema,  interface{},  *Result,  *JsonContext) {

	// Ignore JSON numbers
	if isJSONNumber() {
		return
	}

	// Ignore non strings
	if !isKind(, reflect.String) {
		return
	}

	if internalLogEnabled {
		internalLog("validateString %s", .String())
		internalLog(" %v", )
	}

	 := .(string)

	// minLength & maxLength:
	if .minLength != nil {
		if utf8.RuneCount([]byte()) < int(*.minLength) {
			.addInternalError(
				new(StringLengthGTEError),
				,
				,
				ErrorDetails{"min": *.minLength},
			)
		}
	}
	if .maxLength != nil {
		if utf8.RuneCount([]byte()) > int(*.maxLength) {
			.addInternalError(
				new(StringLengthLTEError),
				,
				,
				ErrorDetails{"max": *.maxLength},
			)
		}
	}

	// pattern:
	if .pattern != nil {
		if !.pattern.MatchString() {
			.addInternalError(
				new(DoesNotMatchPatternError),
				,
				,
				ErrorDetails{"pattern": .pattern},
			)

		}
	}

	// format
	if .format != "" {
		if !FormatCheckers.IsFormat(.format, ) {
			.addInternalError(
				new(DoesNotMatchFormatError),
				,
				,
				ErrorDetails{"format": .format},
			)
		}
	}

	.incrementScore()
}

func ( *subSchema) ( *subSchema,  interface{},  *Result,  *JsonContext) {

	// Ignore non numbers
	if !isJSONNumber() {
		return
	}

	if internalLogEnabled {
		internalLog("validateNumber %s", .String())
		internalLog(" %v", )
	}

	 := .(json.Number)
	,  := new(big.Rat).SetString(string())

	// multipleOf:
	if .multipleOf != nil {
		if  := new(big.Rat).Quo(, .multipleOf); !.IsInt() {
			.addInternalError(
				new(MultipleOfError),
				,
				,
				ErrorDetails{
					"multiple": new(big.Float).SetRat(.multipleOf),
				},
			)
		}
	}

	//maximum & exclusiveMaximum:
	if .maximum != nil {
		if .Cmp(.maximum) == 1 {
			.addInternalError(
				new(NumberLTEError),
				,
				,
				ErrorDetails{
					"max": new(big.Float).SetRat(.maximum),
				},
			)
		}
	}
	if .exclusiveMaximum != nil {
		if .Cmp(.exclusiveMaximum) >= 0 {
			.addInternalError(
				new(NumberLTError),
				,
				,
				ErrorDetails{
					"max": new(big.Float).SetRat(.exclusiveMaximum),
				},
			)
		}
	}

	//minimum & exclusiveMinimum:
	if .minimum != nil {
		if .Cmp(.minimum) == -1 {
			.addInternalError(
				new(NumberGTEError),
				,
				,
				ErrorDetails{
					"min": new(big.Float).SetRat(.minimum),
				},
			)
		}
	}
	if .exclusiveMinimum != nil {
		if .Cmp(.exclusiveMinimum) <= 0 {
			.addInternalError(
				new(NumberGTError),
				,
				,
				ErrorDetails{
					"min": new(big.Float).SetRat(.exclusiveMinimum),
				},
			)
		}
	}

	// format
	if .format != "" {
		if !FormatCheckers.IsFormat(.format, ) {
			.addInternalError(
				new(DoesNotMatchFormatError),
				,
				,
				ErrorDetails{"format": .format},
			)
		}
	}

	.incrementScore()
}
